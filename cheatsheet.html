Javascript Reminders

Dots and Brackets

The important thing to remember here is mixing bracketed and dot notation to get to values in objects.

function Person(name, age) {
    this.name = name;
    this.age = age;
    };

var family = [];

var alice = new Person("alice", 40);
var bob = new Person("bob", 42);
var michelle = new Person("michelle", 8);
var timmy = new Person("timmy", 6);

// Now we can make an array of people

family = [alice, bob, michelle, timmy];


// loop through our new array

for(var i = 0; i <family.length; i++){
    console.log(family[i].name);
};

"This" intro

// here we define our method using "this", before we even introduce bob
var setAge = function (newAge) {
  this.age = newAge;
};
// now we make bob
var bob = new Object();
bob.age = 30;
// and down here we just use the method we already made
bob.setAge = setAge;

// change bob's age to 50 here
bob.setAge(50)

var addAgetoHandedNumber = function(number) {
    this.age = this.age + number;
};

ob = new Object();
ob.age = 43;
console.log(ob.age);
ob.addAgetoHandedNumber = addAgetoHandedNumber;
ob.addAgetoHandedNumber(7);
console.log(ob.age);
